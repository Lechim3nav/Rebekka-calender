  });
}

// Derive a light bg and dark text from a hex colour
function hexToBg(hex){ return hex+'22'; }
function hexToText(hex){
  const r=parseInt(hex.slice(1,3),16), g=parseInt(hex.slice(3,5),16), b=parseInt(hex.slice(5,7),16);
  const lum=(0.299*r+0.587*g+0.114*b)/255;
  return lum>0.55 ? '#1A1D2E' : hex.replace(/^#/,'').length===6 ? darken(hex) : '#fff';
}
function darken(hex){
  let r=parseInt(hex.slice(1,3),16), g=parseInt(hex.slice(3,5),16), b=parseInt(hex.slice(5,7),16);
  r=Math.max(0,r-80); g=Math.max(0,g-80); b=Math.max(0,b-80);
  return '#'+[r,g,b].map(x=>x.toString(16).padStart(2,'0')).join('');
}

function eventStyle(members){
  if(members.length===1){
    const hex=MEMBERS[members[0]].hex;
    return {bg:hexToBg(hex),text:darken(hex),border:hex+'55',colors:[hex],multi:false};
  }
  const colors=members.map(k=>MEMBERS[k].hex);
  const pct=100/colors.length;
  const stops=colors.flatMap((c,i)=>[`${c}38 ${Math.round(i*pct)}%`,`${c}38 ${Math.round((i+1)*pct)}%`]).join(',');
  return {bg:`linear-gradient(135deg,${stops})`,text:darken(MEMBERS[members[0]].hex),border:colors[0]+'55',colors,multi:true};
}

// Returns effective fields for selected members (union, sorted by first member)
function getEffectiveFields(){
  // If a single member is selected, use their field config
  // If multiple, use the union of visible fields in the order of first selected member
  const sel=[...selectedMembers].filter(m=>m!=='family');
  const primary=sel[0]||'mum';
  return MEMBERS[primary].fields.filter(f=>f.show);
}

function buildAddForm(){
  const container=document.getElementById('addFormFields');
  container.innerHTML='';
  const fields=getEffectiveFields();
  const FIELD_HTML={
    name:()=>`<div class="field"><label>${t('fName')}</label><input type="text" id="newName" placeholder="${t('placeholderName')}"></div>`,
    date:()=>`<div class="field"><label>${t('fDate')}</label><input type="date" id="newDate" value="${TODAY_STR}"></div>`,
    time:()=>`<div class="field"><label>${t('fTime')}</label><input type="time" id="newTime"></div>`,
    who: ()=>{
      const btns=Object.entries(MEMBERS).map(([k,m])=>{
        const sel=selectedMembers.has(k);
        const style=sel?`border-color:${m.hex};background:${hexToBg(m.hex)};color:${darken(m.hex)}`:'';
        return `<button class="member-btn${sel?' selected':''}" data-m="${k}" style="${style}"><span class="dot" style="background:${m.hex}"></span>${m.label}</button>`;
      }).join('');
      return `<div class="field"><label>${t('fWho')}</label><div class="member-picker" id="memberPicker">${btns}</div></div>`;
    },
    picto:()=>{
      const opts=PICTOS.map(p=>`<button class="picto-opt${p.e===selectedPicto?' selected':''}" data-e="${p.e}" title="${p.l}">${p.e}</button>`).join('');
      return `<div class="field"><label>${t('fPicto')}</label><div class="picto-picker" id="pictoPicker">${opts}</div></div>`;
    },
  };
  // Always show WHO field first (it controls which other fields are relevant), but respect order
  fields.forEach(f=>{ if(FIELD_HTML[f.id]) container.innerHTML+=FIELD_HTML[f.id](); });

  // Re-attach member picker events
  container.querySelectorAll('.member-btn').forEach(btn=>{ btn.addEventListener('click',()=>{ toggleMember(btn.dataset.m); buildAddForm(); }); });
  // Re-attach picto events
  container.querySelectorAll('.picto-opt').forEach(btn=>{ btn.addEventListener('click',()=>{ selectedPicto=btn.dataset.e; buildAddForm(); }); });
}

function toggleMember(m){
  if(m==='family'){
    selectedMembers.clear(); selectedMembers.add('family');
  } else {
    selectedMembers.delete('family');
    if(selectedMembers.has(m)) selectedMembers.delete(m); else selectedMembers.add(m);
  }
}

function buildPersonPanel(k,m){

      <span class="field-config-icon">${FIELD_ICONS[f.id]||'•'}</span>
      <span class="field-config-name">${fieldKeys[f.id]||f.id}</span>
      <label class="toggle">

        <span class="toggle-slider"></span>
      </label>
    </div>`).join('');

  // Photo preview
  const photoHtml = draft.photo
    ? `<div class="photo-preview-wrap">
         <img class="photo-preview-img" src="${draft.photo}" alt="PHOTO">
         <button class="photo-remove-btn" id="photoRemove-${k}" title="REMOVE">✕</button>
       </div>`
    : `<div class="photo-preview-wrap">
         <div class="photo-preview-initials" style="background:${draft.hex}">${m.label.split(' ').map(w=>w[0]).join('').slice(0,2)}</div>
       </div>`;

  return `
    <div class="settings-section">
      <h3><span class="sec-icon">🖼️</span> ${t('photoLabel')}</h3>
      <div class="photo-section">
        ${photoHtml}
        <div class="photo-actions">
          <label class="photo-upload-label" for="photoInput-${k}">
            📷 ${t('uploadPhoto')}
            <input class="photo-upload-input" type="file" id="photoInput-${k}" accept="image/*">
          </label>
          <div class="photo-hint">${t('photoHint')}</div>
        </div>
      </div>
    </div>
    <div class="settings-section">
      <h3><span class="sec-icon">🎨</span> ${t('colourLabel')}</h3>
      <div class="colour-row">
        <div class="colour-swatch-grid">${swatches}</div>

function attachPanelEvents(panel,k){
  const draft=settingsDraft[k];

  // Colour swatches
  panel.querySelectorAll('.colour-swatch').forEach(sw=>{
    sw.addEventListener('click',()=>{
      draft.hex=sw.dataset.c;
      panel.querySelectorAll('.colour-swatch').forEach(s=>s.classList.toggle('selected',s.dataset.c===draft.hex));
      panel.querySelector('.colour-hex-input').value=draft.hex;
      panel.querySelector('.colour-preview').style.background=draft.hex;
      buildPersonTabs();
    });
  });

  // Hex input
  const hexIn=panel.querySelector('.colour-hex-input');
  const prev=panel.querySelector('.colour-preview');
  hexIn.addEventListener('input',()=>{
    const v=hexIn.value.trim();
    if(/^#[0-9A-Fa-f]{6}$/.test(v)){
      draft.hex=v;
      prev.style.background=v;
      panel.querySelectorAll('.colour-swatch').forEach(s=>s.classList.toggle('selected',s.dataset.c===v));
      buildPersonTabs();
    }
  });

  // ── Photo upload ──────────────────────────────────────────
  const photoInput = panel.querySelector(`#photoInput-${k}`);
  if (photoInput) {
    photoInput.addEventListener('change', () => {
      const file = photoInput.files[0];
      if (!file) return;
      const reader = new FileReader();
      reader.onload = e => {
        draft.photo = e.target.result;
        // rebuild panel to show new preview
        buildPersonPanels();
      };
      reader.readAsDataURL(file);
    });
  }
  const removeBtn = panel.querySelector(`#photoRemove-${k}`);
  if (removeBtn) {
    removeBtn.addEventListener('click', () => {
      draft.photo = null;
      buildPersonPanels();
    });
  }

  // Toggle visibility
  panel.querySelectorAll('.toggle input').forEach(cb=>{
    cb.addEventListener('change',()=>{
      const f=draft.fields.find(x=>x.id===cb.dataset.field);
      if(f) f.show=cb.checked;
    });
  });

  // ── Pointer-based drag-to-reorder ─────────────────────────
  const list = panel.querySelector(`#fieldList-${k}`);
  let dragSrc = null;

  function getRows(){ return [...list.querySelectorAll('.field-config-row')]; }

  function rowFromPoint(x, y){
    return getRows().find(r => {
      const rect = r.getBoundingClientRect();
      return y >= rect.top && y <= rect.bottom;
    }) || null;
  }

  list.querySelectorAll('.field-config-row').forEach(row => {
    row.addEventListener('pointerdown', e => {
      dragSrc = row;
      dragSrc.style.opacity = '0.45';
      dragSrc.style.boxShadow = '0 6px 24px rgba(0,0,0,.18)';
      dragSrc.style.cursor = 'grabbing';
      list.setPointerCapture(e.pointerId);
      e.preventDefault();
    });
  });

  list.addEventListener('pointermove', e => {
    if (!dragSrc) return;
    const target = rowFromPoint(e.clientX, e.clientY);
    if (!target || target === dragSrc) return;
    const rows = getRows();
    const srcI = rows.indexOf(dragSrc);
    const tgtI = rows.indexOf(target);
    if (srcI < tgtI) list.insertBefore(dragSrc, target.nextSibling);
    else             list.insertBefore(dragSrc, target);
  });

  list.addEventListener('pointerup', () => {
    if (!dragSrc) return;
    dragSrc.style.opacity = '';
    dragSrc.style.boxShadow = '';
    dragSrc.style.cursor = '';
    // Commit new order to draft
    const newOrder = getRows().map(r => r.dataset.field);
    draft.fields.sort((a, b) => newOrder.indexOf(a.id) - newOrder.indexOf(b.id));
    getRows().forEach((r, i) => r.dataset.idx = i);
    dragSrc = null;
  });
}
